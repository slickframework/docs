229     /**
230      * @param array|Fieldset|Element $elements
231      */
232     protected function _addInput($elements)
233     {
234         if (is_a($elements, 'Slick\Form\FieldsetInterface')) {
235             $this->_addInput($elements->elements->asArray());
236         } else {
237             if (is_array($elements)) {
238                 /** @var Element $element */
239                 foreach ($elements as $element) {
240                     $this->getInputFilter()->add($element->getInput());
241                 }
242             } else {
243                 /** @var Element $element */
244                 $this->getInputFilter()->add($elements->getInput());
245             }
246         }
247     }
